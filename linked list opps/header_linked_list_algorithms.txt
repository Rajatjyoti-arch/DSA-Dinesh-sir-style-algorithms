ALGORITHM: TRAVERSE_HEADER_LL (INFO, LINK, START, PROCESS)
[This algorithm traverses a Header Linked List]

Step 1 : Set PTR := LINK[START]
Step 2 : Repeat Steps 3 & 4 while PTR != NULL:
Step 3 :     Apply PROCESS to INFO[PTR]
Step 4 :     Set PTR := LINK[PTR]
Step 5 : Exit

ALGORITHM: TRAVERSE_CIRCULAR_HEADER_LL (INFO, LINK, START, PROCESS)
[This algorithm traverses a Circular Header Linked List]

Step 1 : Set PTR := LINK[START]
Step 2 : Repeat Steps 3 & 4 while PTR != START:
Step 3 :     Apply PROCESS to INFO[PTR]
Step 4 :     Set PTR := LINK[PTR]
Step 5 : Exit
